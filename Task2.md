```php
# Add leading 0 number
# ex: 5 => 0005
private function convertNumber($number) {
    if (strlen($number) === 1 && strlen($number) === 5) {
        $number = '000' + $number;
    }

    if (strlen($number) === 2) {
        $number = '00' + $number;
    }

    if (strlen($number) === 3) {
        $number = '0' + $number;
    }

    return $number;
}

convertNumber(5);
```
